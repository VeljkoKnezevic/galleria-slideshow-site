);


@mixin breakpoint($size) {
  @media(min-width: map-get($breakpoints-up, $size)) {
    @content
  }
}

// _functions.scss
@use "sass:math";

@function rem($pixel) {
  @if math.is-unitless($pixel) {
    @return math.div($pixel, 16) + rem;
  } @else {
    @error "dont use units, only numbers";
  }
}

@function em($pixel) {
  @if math.is-unitless($pixel) {
    @return math.div($pixel, 16) + em;
  } @else {
    @error "dont use units, only numbers";
  }
}
